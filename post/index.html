RepoInfo.decode = function(string) {
  if (string == null) return;

  var obj = RepoInfo.getObj("repoInfo.LabelInfo");
  var edtView = document.getElementById("edtView");
  edtView.value = " ";

  var str = RepoInfo.getArray(string);

};

sudo apt remove nodejs
